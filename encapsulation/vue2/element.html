# 分页器-pagination

  • 二次封装

component/cdyPagination.vue

<template>
	<div v-if="total !== 0">
		<!-- 分页器 -->
		<el-pagination
			v-bind="$attrs"
			prev-text="上一页"
			next-text="下一页"
			background
			:total="total"
			:layout="layout"
			:page-sizes="pageSizes"
			:current-page.sync="page"
			:page-size.sync="limit"
			@size-change="handleChange"
			@current-change="handleChange"
		/>
	</div>
</template>

<script>
export default {
	name: 'Pagination',
	props: {
		// 当前页
		currentPage: {
			type: Number,
			default: 1,
		},
		// 当前条数
		pageSize: {
			type: Number,
			default: 20,
		},
		// 总条数
		total: {
			type: Number,
			default: 0,
		},
		// 模块化
		background: {
			type: Boolean,
			default: true,
		},
		// 选择每页展示数量
		pageSizes: {
			type: Array,
			default: () => [10, 20, 50, 100],
		},
		// 组件布局
		layout: {
			type: String,
			default: 'total, prev, pager, next, jumper',
		},
	},
	computed: {
		// 当前页
		page: {
			get() {
				return this.currentPage
			},
			set(val) {
				this.$emit('update:currentPage', val)
			},
		},
		limit: {
			get() {
				return this.pageSize
			},
			set(val) {
				this.$emit('update:pageSize', val)
			},
		},
	},
	methods: {
		// 两个方法合并到一起使用
		handleChange() {
			this.$emit('pagination', {
				currentPage: this.currentPage,
				pageSize: this.pageSize,
			})
		},
	},
}
</script>

# 表格-table

二次封装

<template>
	<div class="cdyTable">
		<el-table :data="tableData" highlight-current-row style="width: 100%;" border :header-cell-style="{background: '#FAFAFA', color: '#333'}">
			<el-table-column v-if="typeIdex" label="序号" type="index" width="90" align="center" />

			<template v-for="item in tableTitle">
				<el-table-column
					v-if="item.prop === 'operation'"
					align="center"
					:key="item.prop"
					:column-key="item.prop"
					:width="`${item.btn.length}20`"
				>
					<template slot="header">
						<div class="headerTit">
							{{ item.title }}
						</div>
					</template>

					<!-- 按钮 -->
					<template slot-scope="scope">
						<el-button
							v-for="(btnItem, index) in item.btn"
							:key="index"
							type="primary"
							size="small"
							@click="handleTable(btnItem.scene, scope.row, scope.$index)"
						>
							{{ btnItem.text }}
						</el-button>
					</template>
				</el-table-column>

				<!-- 正常显示的内容 -->
				<el-table-column v-else align="center" :key="item.prop" :column-key="item.prop">
					<template slot="header">
						<div class="headerTit">{{ item.title }}</div>
					</template>
					<template slot-scope="scope">
						<div class="table-content">
							{{ scope.row[item.prop] }}
						</div>
					</template>
				</el-table-column>
			</template>
			<div class="nolist" slot="empty" v-if="!empty">
				<p>暂无数据</p>
			</div>
		</el-table>
	</div>
</template>
<script>
export default {
	props: {
		// 是否要序号
		typeIdex: {
			type: Boolean,
			default: false,
		},
		// 是否要 无数据展示提示语
		empty: {
			type: Boolean,
			default: true,
		},
		// table 的标题数组
		tableTitle: {
			type: Array,
			default: () => [],
			require: true,
		},
		// table 的数据
		tableData: {
			type: Array,
			default: () => [],
			require: true,
		},
	},
	methods: {
		// 表格操作
		handleTable(scene, row, index) {
			// scene 方法
			// row   数据源
			// index table 索引
			this.$emit('handleTable', scene, row, index)
		},
	},
}
</script>
